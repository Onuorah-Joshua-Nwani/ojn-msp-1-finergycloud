IRR Calculation Algorithm
function calculateIRR(cashFlows, guess = 0.1) {
  const maxIterations = 1000;
  const tolerance = 1e-10;
  
  let rate = guess;
  
  for (let i = 0; i < maxIterations; i++) {
    const npv = calculateNPV(cashFlows, rate);
    
    if (Math.abs(npv) < tolerance) {
      return rate;
    }
    
    const derivativeNPV = calculateDerivativeNPV(cashFlows, rate);
    
    if (Math.abs(derivativeNPV) < tolerance) {
      break;
    }
    
    // Newton-Raphson method
    rate = rate - npv / derivativeNPV;
  }
  
  return rate;
}

function calculateNPV(cashFlows, rate) {
  return cashFlows.reduce((npv, cf, t) => {
    return npv + cf / Math.pow(1 + rate, t);
  }, 0);
}

function calculateDerivativeNPV(cashFlows, rate) {
  return cashFlows.reduce((dnpv, cf, t) => {
    if (t === 0) return dnpv;
    return dnpv - (t * cf) / Math.pow(1 + rate, t + 1);
  }, 0);
}
